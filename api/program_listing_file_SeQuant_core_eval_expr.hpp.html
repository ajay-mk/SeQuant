  [[nodiscard]] bool is_tensor() const noexcept;

  [[nodiscard]] bool is_constant() const noexcept;

  [[nodiscard]] bool is_variable() const noexcept;

  [[nodiscard]] Tensor const& as_tensor() const noexcept;

};

}  // namespace sequant
